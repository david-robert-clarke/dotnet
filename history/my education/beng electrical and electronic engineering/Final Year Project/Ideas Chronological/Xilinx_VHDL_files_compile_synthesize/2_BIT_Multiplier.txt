LIBRARY IEEE;
USE IEEE.STD_LOGIC_1164.ALL;
USE IEEe.STD_LOGIC_ARITH.ALL;
--
ENTITY MUL2 IS
PORT (A, B : IN unsigned (1 downto 0); Z : OUT unsigned (3 downto 0));
END ENTITY MUL2;

ARCHITECTURE BEHAVIOURAL OF MUL2 IS
BEGIN
	Z <= A * B;
	
END ARCHITECTURE BEHAVIOURAL;
